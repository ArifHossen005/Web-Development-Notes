# Update  

single --- UPDATE `students` SET `name`='Arif' WHERE `id`=5;
Multiple ---

UPDATE `students`
SET `name` = CASE
    WHEN `id` = 5 THEN 'Arif'
    WHEN `id` = 4 THEN 'Karim'
    WHEN `id` = 7 THEN 'Rahim'
    ELSE `name` -- Keeps the existing value if `id` is not 5, 4, or 7
END
WHERE `id` IN (5, 4, 7);
